File to edit: Create src/composables/useRuleExplainer.ts
What to add: A composable that takes the current CodeMirror editor value as a Ref<string> and returns a Ref<string> explanation, populated via a debounced POST to /api/explain-rule. 

// src/composables/useRuleExplainer.ts
import { ref, watch } from "vue";
import { useDebounceFn } from "@vueuse/core";

export function useRuleExplainer(rule: Ref<string>) {
  const explanation = ref("");
  const loading = ref(false);

  const explain = useDebounceFn(async (r: string) => {
    if (!r.includes("pl.col")) { explanation.value = ""; return; }
    loading.value = true;
    const res = await fetch("/api/explain-rule", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ rule: r }),
    });
    const data = await res.json();
    explanation.value = data.explanation;
    loading.value = false;
  }, 300);

  watch(rule, explain);
  return { explanation, loading };
}

Wire this into your editor component and display explanation in a subtle sidebar. Add the Express /api/explain-rule handler with a 3-line LLM call to get started.
