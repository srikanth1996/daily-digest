What to add: A columnNameCompletion source that reads the active DataFrame schema (column names + types) from Pinia store and surfaces them as autocomplete options when the cursor is inside a pl.col("…") call. 

import { autocompletion, CompletionContext } from "@codemirror/autocomplete";

function columnCompletion(columns: string[]) {
  return (ctx: CompletionContext) => {
    const match = ctx.matchBefore(/pl\.col\(["'][^"']*$/);
    if (!match) return null;
    return {
      from: match.from + 8,  // after pl.col("
      options: columns.map(col => ({ label: col, type: "variable" }))
    };
  };
}

// In your extensions array:
autocompletion({ override: [columnCompletion(store.schemaColumns)] })

This is a zero-dependency improvement (no LLM needed) that immediately reduces typos in column names — the #1 source of validation rule errors in practice.
